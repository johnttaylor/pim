24/** This abstract class define 'equipments' interface to a "stage". The "stage"
25 is responsible for determine what HVAC outputs are on based on the current
26 loading.
27 */
28class StageApi
29{
30public:
31 /** This method is used to activate the stage (i.e. begin actively conditioning
32 the air) when it is FIRST stage (of possibly many) for the current
33 operating mode.
34 */
35 virtual void requestOn( Storm::Component::Control::Equipment::Args_T& args, bool startInOnCycle=true ) noexcept = 0;
36
37 /** This method is used to transition from current stage to the next/higher
38 stage. The method should be called on the current active stage. The
39 'nextStage' is notified when it is becomes the active stage.
40 */
41 virtual void requestAsSupplement( Storm::Component::Control::Equipment::Args_T& args, StageApi& nextStage, bool startNextStageInOnCycle=true ) noexcept = 0;
42
43 /** This method is used to turn off (deactivate) the stage. If the stage
44 is the first stage, then the equipment is turned off. If the stage is
45 is not the first, then previous active stage is notified when it
46 becomes the active stage.
47 */
48 virtual void requestOff( Storm::Component::Control::Equipment::Args_T& args, bool startLowerStageInOnCycle=true ) noexcept = 0;
49
50 /** This method is used to 'reset' the StageApi's internal FSM to the off
51 state. The intended use for this method is turn off the stage when system
52 transitions to the off operating mode and/or to an operating mode that is
53 not appropriate for the stage's operating mode.
54
55 NOTE: For this scenario - the new operating mode is expected/required
56 to ensure the HVAC outputs are in the proper state (i.e. clean-up
57 previous the mode's outputs) - so that stage being requested to
58 OFF does not set/clear any HVAC outputs.
59 */
60 virtual void requestModeToOff() noexcept = 0;
61
62
63public:
64 /** This method is used execute/run the stage. This method should be called

66 be called for all stages (not just the current stage).

68 virtual void execute( Storm::Component::Control::Equipment::Args_T& args ) noexcept = 0;

72 /** This method is used 're-configure' stage after it has been constructed.

75 */
76 virtual void reconfigure( unsigned systemStageIndex, unsigned outputStageIndex ) noexcept = 0;
77
78
79public:
80 /** This method returns true if the stage is providing capacity AND is
81 consider to be the active/highest operating stage.
82
83 Note: The return value is independent of whether the stage is in On or
84 Off Cycle (i.e. if the stage is active/highest stage and current
85 executing on off cycle the method returns true).
86 */
87 virtual bool isActive() const noexcept = 0;
88
89 /** This method returns true if the stage is providing capacity but IS NOT

92 virtual bool isSupplementing() const noexcept = 0;
93
94 /** This method returns true if the stage is in the off state, i.e. the
95 stage is NOT being used in any way to provide capacity.
96
97 Note: This method does NOT report the on/off cycle state/status of
98 stage.
99 */
100 virtual bool isOff() const noexcept = 0;
101
102 /** This method returns true if the current stage is in the process of
103 transitioning to being the active stage (from a lower stage).
104
105 Note: When this method returns true, the method isActive() also returns
106 true.
107 */
108 virtual bool isTransitioningFromLowerStage() const noexcept = 0;
109
110 /** This method returns true if the current stage is in the process of

113 Note: When this method returns true, the method isActive() also returns
114 true.
115 */
116 virtual bool isTransitioningBackToLowerStage() const noexcept = 0;
117
118 /** This method returns true if the current stage is in an on-cycle
119
120 Note: When this method returns true, the method isActive() returns true

122 */
123 virtual bool isOnCycle() const noexcept = 0;
124
125 /** This method returns true if the current stage is in an off-cycle

129 */
130 virtual bool isOffCycle() const noexcept = 0;

141 virtual void notifyAsActiveStage_( Storm::Component::Control::Equipment::Args_T& args, StageApi& previousStage, bool startInOnCycle ) noexcept = 0;
142
