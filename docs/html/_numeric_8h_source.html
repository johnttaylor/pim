147/** This template class extends the Numeric<> class to provide bit operation
148 on the numeric value. The datatype of the numeric MUST be an integer
149 type.
150
151 The underlying storage of the bit array is N bit integers. A side effect of
152 this storage mechanism the bit ordering in the JSON 'val' string is dependent on the
153 target platform's Endian architecture.
154
155 The toJSON()/fromJSON format is:
156 \code
157
158 { name:"<mpname>", type:"<mptypestring>", valid:true|false, seqnum:nnnn, locked:true|false, val:"<bits>" }
159
160 where <bits> is a string of N digits ('1' or '0') where the left most digit is
161 is the MSb of byte[0] and the right most digit is the LSb of byte[N].
162 Whether byte[0] is the MSB or LSB is dependent on the big/little Endian
163 architecture of the target platform.
164
165 For example a 16bit Array (as binary hex: dataword[0]=0x30, dataword[1]=0x09)
166
167 val:"0011000000001001"
168
169 \endcode
170
171 NOTE: All methods in this class ARE thread Safe unless explicitly
172 documented otherwise.
173
174 */
175template <class WORDSIZE, class MPTYPE>
176class BitArray_ : public Numeric<WORDSIZE, MPTYPE>
177{
178protected:
179 /// Constructor. Invalid MP.
180 BitArray_( Cpl::Dm::ModelDatabase& myModelBase, const char* symbolicName )
181 : Numeric<WORDSIZE, MPTYPE>( myModelBase, symbolicName )
182 {
183 }
184
185 /// Constructor. Valid MP. Requires an initial value
186 BitArray_( Cpl::Dm::ModelDatabase& myModelBase, const char* symbolicName, WORDSIZE initialValue )
187 : Numeric<WORDSIZE, MPTYPE>( myModelBase, symbolicName, initialValue )
188 {
189 }
190
191public:
192 /// Atomic operation to set the zero indexed bit to a 1.
193 inline uint16_t setBit( uint8_t bitPosition, Cpl::Dm::ModelPoint::LockRequest_T lockRequest = Cpl::Dm::ModelPoint::eNO_REQUEST ) noexcept
194 {
195 Cpl::Dm::ModelPointCommon_::m_modelDatabase.lock_();
196 uint16_t result = Numeric<WORDSIZE, MPTYPE>::write( Numeric<WORDSIZE, MPTYPE>::m_data | ( 1 << bitPosition ), lockRequest );
197 Cpl::Dm::ModelPointCommon_::m_modelDatabase.unlock_();
198 return result;
199 }
200
201 /// Atomic operation to set the zero indexed bit to a 0.
202 inline uint16_t clearBit( uint8_t bitPosition, Cpl::Dm::ModelPoint::LockRequest_T lockRequest = Cpl::Dm::ModelPoint::eNO_REQUEST ) noexcept
203 {
204 Cpl::Dm::ModelPointCommon_::m_modelDatabase.lock_();
205 uint16_t result = Numeric<WORDSIZE, MPTYPE>::write( Numeric<WORDSIZE, MPTYPE>::m_data & ( ~( 1 << bitPosition ) ), lockRequest );
206 Cpl::Dm::ModelPointCommon_::m_modelDatabase.unlock_();
207 return result;
208 }
209
210 /// Atomic operation to toggle the zero indexed bit.
211 inline uint16_t flipBit( uint8_t bitPosition, Cpl::Dm::ModelPoint::LockRequest_T lockRequest = Cpl::Dm::ModelPoint::eNO_REQUEST ) noexcept
212 {
213 Cpl::Dm::ModelPointCommon_::m_modelDatabase.lock_();
214 uint16_t result = Numeric<WORDSIZE, MPTYPE>::write( Numeric<WORDSIZE, MPTYPE>::m_data ^ ( 1 << bitPosition ), lockRequest );
215 Cpl::Dm::ModelPointCommon_::m_modelDatabase.unlock_();
216 return result;
217 }
218
219
220public:
221 /// Atomic operation to clear ONLY the bits as specified by the bit mask.
222 inline uint16_t clearBitsByMask( uint16_t bitMask, Cpl::Dm::ModelPoint::LockRequest_T lockRequest = Cpl::Dm::ModelPoint::eNO_REQUEST ) noexcept
223 {
224 Cpl::Dm::ModelPointCommon_::m_modelDatabase.lock_();
225 uint16_t result = Numeric<WORDSIZE, MPTYPE>::write( Numeric<WORDSIZE, MPTYPE>::m_data & ~( bitMask ), lockRequest );
226 Cpl::Dm::ModelPointCommon_::m_modelDatabase.unlock_();
227 return result;
228 }
229
230 /// Atomic operation to set the bits specified by the bit mask
231 inline uint16_t setBitsByMask( uint16_t bitMask, Cpl::Dm::ModelPoint::LockRequest_T lockRequest = Cpl::Dm::ModelPoint::eNO_REQUEST ) noexcept
232 {
233 Cpl::Dm::ModelPointCommon_::m_modelDatabase.lock_();
234 uint16_t result = Numeric<WORDSIZE, MPTYPE>::write( Numeric<WORDSIZE, MPTYPE>::m_data | bitMask, lockRequest );
235 Cpl::Dm::ModelPointCommon_::m_modelDatabase.unlock_();
236 return result;
237 }
238
239 /// Atomic operation to flip/toggle ONLY the bits as specified the bit mask
240 inline uint16_t flipBitsByMask( uint16_t bitMask, Cpl::Dm::ModelPoint::LockRequest_T lockRequest = Cpl::Dm::ModelPoint::eNO_REQUEST ) noexcept
241 {
242 Cpl::Dm::ModelPointCommon_::m_modelDatabase.lock_();
243 uint16_t result = Numeric<WORDSIZE, MPTYPE>::write( Numeric<WORDSIZE, MPTYPE>::m_data ^ bitMask, lockRequest );
244 Cpl::Dm::ModelPointCommon_::m_modelDatabase.unlock_();
245 return result;
246 }
