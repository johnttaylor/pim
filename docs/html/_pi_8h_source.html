28/** This concrete class implements a simple (P)roportional (I)ntegral
29 controller that only allows the integral term to be a positive number. This
30 class is simplification of a basic PID Controller (see https://en.wikipedia.org/wiki/PID_controller)
31
32 The class has the following 'integral wind-up' protections:
33
34 o Allows many external entities to explicitly inhibit the integral term.
35
36 o The output of PI is clamped based on the configuration parameter
37 piConstants.maxPvOut. When the output is clamped the integral term is
38 inhibited.
39
40 o The integral term is clamped such that integral term by itself
41 (i.e. with zero error) does not exceed the maximum configured
42 output of the PI.
43 */
44class Pi : public Base
45{
46public:
47 /// Input Model Points
48 struct Input_T
49 {
50 Cpl::Dm::Mp::Bool* pulseResetPi; //!< Triggers a reset-the-PI-controller request
51 Cpl::Dm::Mp::Float* idtDeltaError; //!< The delta error (in degrees F) between the current IDT the 'active' setpoint
52 Storm::Dm::MpSystemConfig* systemConfig; //!< Current system configuration based on equipment and current operating mode
53 Cpl::Dm::Mp::RefCounter* freezePiRefCnt; //!< Reference Counter: When greater the zero the PI output value is 'frozen' (i.e. won't change value) and the internal integral term is not updated/changed
54 Cpl::Dm::Mp::RefCounter* inhibitfRefCnt; //!< Reference Counter: When greater the zero the internal integral term is not update/changed.
55 };
56
57
58 /// Output Model Points
59 struct Output_T
60 {
61 Cpl::Dm::Mp::Float* pvOut; //!< Output of the PI Controller. This is unit-less positive number that ranges from 0.0 to piConstants.maxPvOut
62 Cpl::Dm::Mp::Float* sumError; //!< An internal intermediate value/variable that represents the integral term of the PI
63 Cpl::Dm::Mp::Bool* pvInhibited; //!< This flag is true if the PI Controller's integral is currently inhibited. Note: This flag includes any internal inhibiting of the integral term as well as being set true when the PI 'frozen'
64 };
