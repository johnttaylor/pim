31/** This function performs a blocking wait by monitoring a Model Point for
32 a specific value. The method waits a specified amount of time BEFORE
33 checking for the expected value. The method returns true if the specified
34 MP's values matches the expected value; else the method will eventually time
35 out and return false.
36 */
37template<class MPTYPE, class VALTYPE>
38bool minWaitOnModelPoint( MPTYPE& src, VALTYPE expectedVal, unsigned long minWait, VALTYPE* lastVal=nullptr, unsigned long maxWaitMs=10000, unsigned long waitDelayMs=10 )
39{
40 Cpl::System::Api::sleep( minWait );
41
42 VALTYPE val;
43 bool valid = src.read( val );
44 while ( maxWaitMs )
45 {
46 if ( valid && val == expectedVal )
47 {
48 // Return the last read value
49 if ( lastVal )
50 {
51 *lastVal = val;
52 }
53 return true;
54 }
55
56 Cpl::System::Api::sleep( waitDelayMs );
57 maxWaitMs -= waitDelayMs;
58 valid = src.read( val );
59 }
60
61 // Return the last read value
62 if ( lastVal )
63 {
64 *lastVal = val;
65 }
66 return false;
67};

109{
