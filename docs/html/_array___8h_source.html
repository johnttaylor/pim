131/** This template class extends the implementation of ArrayBase_ to support
132 the toJSON() and fromJSON_() methods for numeric element types.
133
134 NOTES:
135 1) All methods in this class are NOT thread Safe unless explicitly
136 documented otherwise.
137*/
138template<class ELEMTYPE>
139class NumericArrayBase_ : public ArrayBase_
140{
141protected:
142 /// Constructor: Invalid MP
143 NumericArrayBase_( Cpl::Dm::ModelDatabase& myModelBase,
144 const char* symbolicName,
145 ELEMTYPE* myDataPtr,
146 size_t numElements )

155 */
156 NumericArrayBase_( Cpl::Dm::ModelDatabase& myModelBase,
157 const char* symbolicName,
158 ELEMTYPE* myDataPtr,
159 size_t numElements,
160 ELEMTYPE* srcData )
161 :ArrayBase_( myModelBase, symbolicName, myDataPtr, numElements, sizeof( ELEMTYPE ), srcData )

166public:

169 {

172

174 inline uint16_t write( const ELEMTYPE* srcArray, size_t srcNumElements, size_t dstIndex = 0, Cpl::Dm::ModelPoint::LockRequest_T lockRequest = Cpl::Dm::ModelPoint::eNO_REQUEST ) noexcept
175 {
176 return ArrayBase_::write( srcArray, srcNumElements, dstIndex, lockRequest );
177 }
178
179protected:
180 /// See Cpl::Dm::Point.
181 void setJSONVal( JsonDocument& doc ) noexcept
182 {
183 JsonObject obj = doc.createNestedObject( "val" );
184 obj["start"] = 0;
185 JsonArray arr = obj.createNestedArray( "elems" );
186 ELEMTYPE* elemPtr = (ELEMTYPE*) m_dataPtr;
187 for ( size_t i = 0; i < m_numElements; i++ )
188 {
189 arr.add( elemPtr[i] );
190 }
191 }
192
193public:
194 /// See Cpl::Dm::Point.
195 bool fromJSON_( JsonVariant& src, Cpl::Dm::ModelPoint::LockRequest_T lockRequest, uint16_t& retSequenceNumber, Cpl::Text::String* errorMsg ) noexcept
196 {
197 // Check for object
198 if ( src.is<JsonObject>() == false )
199 {
200 if ( errorMsg )
201 {
202 *errorMsg = "'val' key/value pair is NOT an JSON object";
203 }
204 return false;
205 }
206
207 // Check for embedded array
208 JsonArray elems = src["elems"];
209 if ( elems.isNull() )
210 {
211 if ( errorMsg )
212 {
213 *errorMsg = "'val' key/value pair is missing the embedded 'elems' array";
214 }
215 return false;
216 }
217
218 // Get starting index (note: if not present a default of '0' will be returned)
219 size_t startIdx = src["start"] | 0;
220
221 // Check for exceeding array limits
222 size_t numElements = elems.size();
223 if ( numElements + startIdx > m_numElements )
224 {
225 if ( errorMsg )
226 {
227 errorMsg->format( "Number of array elements ([%lu+%lu)] exceeds the MP's element count (%lu)", startIdx, numElements, m_numElements );
228 }
229 return false;
230 }
231
232 // Update the Model Point in 'M' elements at a time (helps to reduce 'noise' on the MP's sequence number)
233 size_t offset = 0;
234 while ( numElements )
235 {
236 // Attempt to parse the value key/value pair (as a simple numeric)
237 ELEMTYPE tempArray[OPTION_CPL_DM_MP_ARRAY_TEMP_ARRAY_NUM_ELEMENTS];
238 size_t idx;
239 for ( idx = 0; idx < numElements && idx < OPTION_CPL_DM_MP_ARRAY_TEMP_ARRAY_NUM_ELEMENTS; idx++ )
240 {
241 // Is the element syntacticly correct?
242 if ( elems[idx].is<ELEMTYPE>() == false )
243 {
244 if ( errorMsg )
245 {
246 errorMsg->format( "Failed parsing element[%lu]. Content of the MP is suspect!", offset );
247 }
248 return false;
249 }
250 tempArray[idx] = elems[idx + offset].as<ELEMTYPE>();
251 }
252 retSequenceNumber = ArrayBase_::write( tempArray, idx, startIdx + offset, lockRequest );
253 offset += idx;
254 numElements -= idx;
255 }
256
257 return true;
258 }
259};
