26/** This class implements a brute force approach to initializing a 'pico
27 mutex' when the Cpl C++ library is initialized. This is to support
28 statically allocated/created Cpl::System::Mutex instances. The issue here
29 is that 'pico mutex' need to be initialized AFTER main() is called. The
30 Pico-SDK solution for statically allocating mutex instances is C solution
31 that does not work inside the Cpl::System::Mutex object.
32 */
33class Mutex_T : public Cpl::System::StartupHook_
34{
35public:
36 /** Constructor.
37 Note: The registration is assumed to be done from a static allocated
38 instance (i.e. before main() and Cpl::System::Api::initialize() is
39 called). However, registering AFTER wards is a don't care because
40 the 'start-up list' is only used during the call to Cpl::System::Api::initialize()
41 */
42 Mutex_T() :StartupHook_( eSYSTEM ), m_rp2040Mutex(nullptr) {}
43
44
45protected:
46 /// This method is called when Cpl::System::Api::initialize() executes
47 void notify( InitLevel_T init_level )
48 {
49 m_rp2040Mutex = new recursive_mutex_t;

51 {
52 recursive_mutex_init( m_rp2040Mutex );
53 }
54 }
55
56public:
57 /// Platform Mutex instance. Note: Made public to simply access from the Cpl::System::Mutex class
