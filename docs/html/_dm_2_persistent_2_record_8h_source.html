31/** This mostly concrete class implements the Cpl::Persistent::Record interface
32 where a Record instance contains the data from N model points. A final
33 child class is needed to provide the specifics of 'resetting' the Record's
34 data.
35 */
36class Record: public Cpl::Persistent::Record, public Cpl::Persistent::Payload, public FlushRequest, public EraseRequest
37{
38public:
39 /** This data structure associates a Data Model subscriber instance with a

41 */
42 typedef struct

44 Cpl::Dm::ModelPoint* mpPtr; //!< Pointer to a Model Point

46 } Item_T;

48public:
49 /** Constructor. The 'itemList' is variable length array where the last item
50 in the list MUST be a 'null' entry, e.g. {0,0}. The application is
51 responsible for ALLOCATING all of the Model Point and Subscriber
52 instances in the itemList.
53
54 If the schema indexes does not match when reading data from persistent
55 storage, THEN the upgrade method is called.record data is defaulted. If the schema minor index from
56 persistent storage does is less than 'schemaMinorIndex' argument, the
57 the record is data loaded and the upgrade() method is called. If the
58 schema minor index from persistent storage is greater than the
59 'schemaMinorIndex' argument, THEN the record data is defaulted.
60 */
61 Record( Item_T itemList[], Cpl::Persistent::Chunk& chunkHandler, uint8_t schemaMajorIndex, uint8_t schemaMinorIndex ) noexcept;

106 record to their default values. This method is called when there is
107 NO valid data when reading the record's data from persistence storage.

109 When the method returns true, the 'reset data' is written to persistence
110 storage; else the persistence storage is NOT updated and NO further updates
111 of the persistence storage will be allowed.
112 */
113 virtual bool resetData() noexcept = 0;
114
115 /** This method is called when the stored record data has different
116 schema indexes. It is up to the child record class on what to do when
117 the schema is different. The default behavior is default all of the
118 data.
119
120 The 'src' argument contains the record data (as stored in persistent
121 storage). The child class can make use of the data as needed. Note:
122 the data in 'src' is AFTER the schema indexes fields.
123
124 The method should return true if the child class has successfully
125 processed in the incoming data. Returning false will cause the
126 record data to be defaulted;
127 */
128 virtual bool schemaChange( uint8_t previousSchemaMajorIndex,
129 uint8_t previousSchemaMinorIndex,
130 const void* src,
131 size_t srcLen ) noexcept { return false; }
