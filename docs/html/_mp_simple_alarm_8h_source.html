26/** This class provides a concrete implementation for a simple Alarm that has
27 following attributes:
28 - Requires some action (other than acknowledgment) to clear the Alarm's root cause
29 - Tracks the 'user acknowledgeable' state
30 - Provides an indication if the Alarm is critical, i.e. whether or not the Alarm cause the system be forced off
31
32 The toJSON() method is a read/modify operation, i.e. omitted key/value
33 fields in the 'val' object are NOT updated.
34
35 The toJSON()/fromJSON format is:
36 \code
37
38 { name:"<mpname>", type:"<mptypestring>", valid:true|false, seqnum:nnnn, locked:true|false, val:{"active":"true|false, "ack":true|false, "critical":true|false} }
39
40 \endcode
41
42
43 NOTE: All methods in this class ARE thread Safe unless explicitly
44 documented otherwise.
45 */
46class MpSimpleAlarm : public Cpl::Dm::ModelPointCommon_
47{
48public:

53 bool active; //!< When set to true, the alarm is 'active'

55 bool critical; //!< When set to true, the system is/was forced to the its 'off state'
56 };
