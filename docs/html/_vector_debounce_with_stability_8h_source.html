87 /// See Filter
88 Driver::Imu::Vector<T> filterValue( const Driver::Imu::Vector<T>& rawValue )
89 {
90 // Allow the first value to come through unfiltered
91 if ( m_haveBeenCalledAtLeastOnce )
92 {
93 m_lastDebounced = rawValue;
94 m_lastStable = rawValue;
95 m_stable = true;
96 }
97
98 // Apply the debounce algorithm
99 else
100 {
101 // Debounce the incoming raw values
102 bool isXEqual = equalEnough( rawValue.x, m_lastRaw.x, m_tolerance );
103 bool isYEqual = equalEnough( rawValue.y, m_lastRaw.y, m_tolerance );
104 bool isZEqual = equalEnough( rawValue.z, m_lastRaw.z, m_tolerance );
105 if ( !isXEqual || !isYEqual || !isZEqual )
106 {
107 m_debounceCounter = 0;
108 m_stabilityCounter = 0;
109 m_stable = false;
110 }
111 else
112 {
113 if ( ++m_debounceCounter >= m_debounceLimit )
114 {
115 m_lastDebounced = rawValue;
116 m_debounceCounter = m_debounceLimit;
117
118 // Check for stability
119 if ( ++m_stabilityCounter >= m_stableLimit )
120 {
121 m_stable = true;
122 m_lastStable = rawValue;
123 m_stabilityCounter = m_stableLimit;
124 }
125 }
126 }
127 m_lastRaw = rawValue;
128 }
129
130 return m_onlyReturnStableValues ? m_lastStable : m_lastDebounced;
131 }
132
133
134protected:
135 /// Helper function
136 static bool equalEnough( T a, T b, T tolerance )
137 {
138 if ( a <= b )
139 {
140 if ( a >= b - tolerance )
141 {
142 return true;
143 }
144 }
145 else
146 {
147 if ( a <= b + tolerance )
148 {
149 return true;
150 }
151 }
