19/** This class simulates a "system" whose state is affected
20 by its environment and any number of control systems
21 that may change the environment over time. The design of
22 this class was inspired by the electrical theory surrounding
23 the charging and discharging of a capacitor by a voltage
24 source (potential) through a resistance. The environment
25 in exactly the same way as the various controlling sources,
26 that is as a potential and a series resistance. Although
27 this model is based upon electrical engineering, it is
28 essentially unit-less and may be applied to similarly behaving
29 systems (e.g. thermal systems).
30 */
31class SimSystem
32{
33private:
34 /** Contains the time in seconds between system
35 updates.
36 */
37 const double _tickPeriodInSeconds;
38
39 /** The potential of the system after the previous
40 system update. This value is used as the starting
41 point for the next system update calculation,
42 and is modified appropriately at that time.
43 */
44 double _previousOutputPotential;
45
46public:
47 /** This value represents the default value of the
48 resistance between the system and the environmental
49 potential.
50 */
51 double _environmentResistance;
52
53 /** This value represents the capacity of the system.
54 E.g. in an electrical system this is the value of
55 the capacitor. It affects the overall "inertia" of
56 the simulated system.
57 */
58 double _capacitance;
59
60 /** This value is used during the start() -> accumulate() ->
61 finish() sequence to hold the accumulated "equivalent"
62 potential of the combined environment and control
63 sources. Its value is calculated in the same way as
64 the "open-circuit voltage" in a Thevenin equivalent
65 circuit.
66 */
67 double _peq;
68
69 /** This value is used during the start() -> accumulate() ->
70 finish() sequence to hold the accumulated "equivalent"
71 resistance of the combined environment and control
72 sources. Its value is calculated in the same way as
73 the "equivalent series resistance" in a Thevenin equivalent
74 circuit.
75 */
76 double _req;
77
78public:
79 /**
80 * tickPeriodInSeconds is the time between each
81 * call to the tick() operations.
82 * controlResistance is the "internal" resistance
83 * of the control input to other network.
84 * environmentResistance is the "internal" resistance
85 * of the environment.
86 * capacitance is the amount of charge that can be
87 * stored in the system.
88 * initialOutputPotential is the initial voltage across
89 * the capacitor/system.
90 */
91 SimSystem( double tickPeriodInSeconds,
92 double environmentResistance,
93 double capacitance,
94 double initialOutputPotential ) noexcept;
95
96 /** This operation is used to start an accumulation cycle.
97 It allows several sources to affect the calculation
98 of the system output over the next cycle. In this
99 variant, the potential is that of the "environment"
100 and the resistance is assumed to be that of the
101 environment as specified by the constructor.
102 */
103 void start( double potential ) noexcept;
104
105 /** This operation is used to start an accumulation cycle.
106 It allows several sources to affect the calculation
107 of the system output over the next cycle. In this
108 variant, the potential is that of the "environment"
109 and the resistance is also that of the "environment"
110 and is used instead of the constant supplied in the
111 constructor.
112 */
113 void start( double potential, double resistance ) noexcept;
114
115 /** This operation is to be invoked zero or more times after
116 one of the start() operations and before the finish()
117 operation. Each invocation accumulates a new potential
118 source into the system calculation. This allows any
119 number of sources, each consisting of a potential and
120 a resistance to affect the next state of the system.
121 */
122 void accumulate( double potential, double resistance ) noexcept;
123
124 /** This operation is issued after one of the start() operations
125 and zero or more accumulate() operations to calculate the
126 new output potential/state of the system, that is returned as
127 a result.
128 */
129 double finish() noexcept;
130
131};
