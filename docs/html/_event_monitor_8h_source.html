134 /// Button event
135 void buttonEventMode( Cpl::Dm::Mp::Bool& mp, Cpl::Dm::SubscriberApi& clientObserver )
136 {
137 bool pressed;
138 if ( mp.readAndSync( pressed, clientObserver ) && pressed )
139 {
140 // Check for Setpoint Selection active
141 if ( m_setPointActive )
142 {
143 toggleSetpoint();
144 return;
145 }
146
147 // Start selection
148 if ( !m_modeActive )
149 {
150 m_modeActive = true;
151 m_fanActive = false;
152 drawFan();
153 }
154
155 // Toggle to the next value
156 else
157 {
158 unsigned curIdx = g_uiMode._to_index();
159 if ( ++curIdx >= Storm::Type::ThermostatMode::_size() )
160 {
161 curIdx = 0;
162 }
163 g_uiMode = Storm::Type::ThermostatMode::_from_index_unchecked( curIdx );
164 }
165
166 m_timer.start( OPTION_STORM_THERMOSTAT_UI_PICO_DISPLAY_DELAY_UPDATES_MS );
167 drawMode( true );
168 }
169 }

171 /// Button event
172 void buttonEventFan( Cpl::Dm::Mp::Bool& mp, Cpl::Dm::SubscriberApi& clientObserver )
173 {
174 bool pressed;
175 if ( mp.readAndSync( pressed, clientObserver ) && pressed )
176 {
177 // Check for Setpoint Selection active
178 if ( m_setPointActive )
179 {
180 toggleSetpoint();
181 return;
182 }
183
184 // Start selection
185 if ( !m_fanActive )
186 {
187 m_modeActive = false;
188 m_fanActive = true;
189 drawMode();
190 }
191
192 // Toggle to the next value
193 else
194 {
195 if ( g_uiFan == +Storm::Type::FanMode::eCONTINUOUS )
196 {
197 g_uiFan = Storm::Type::FanMode::eAUTO;
198 }
199 else
200 {
201 g_uiFan = Storm::Type::FanMode::eCONTINUOUS;
202 }
203 }
204
205 m_timer.start( OPTION_STORM_THERMOSTAT_UI_PICO_DISPLAY_DELAY_UPDATES_MS );
206 drawFan( true );
207 }

210 /// Button event
211 void buttonEventUp( Cpl::Dm::Mp::Bool& mp, Cpl::Dm::SubscriberApi& clientObserver )
212 {
213 bool pressed;
214 if ( mp.readAndSync( pressed, clientObserver ) && pressed )
215 {
216 // Start selection
217 if ( !m_setPointActive )
218 {
219 m_setPointActive = true;
220 m_coolingSetpoingActive = true;
221 m_modeActive = false;
222 m_fanActive = false;

226

228 else
229 {
230 changeSetpoint( +1.0F );
231 }
232
233 m_timer.start( OPTION_STORM_THERMOSTAT_UI_PICO_DISPLAY_DELAY_UPDATES_MS );
234 updateSetpoints();
235 }

274 /// Helper method to update the state/value of the setpoints
275 void updateSetpoints()
276 {
277 if ( m_setPointActive )
278 {
279 if ( m_coolingSetpoingActive )
280 {
281 drawCoolingSetpoint( true );
282 drawHeatingSetpoint();
283 }
284 else
285 {
286 drawCoolingSetpoint();
287 drawHeatingSetpoint( true );
288 }
289 }
290 else
291 {
292 drawCoolingSetpoint();
293 drawHeatingSetpoint();
294 }
295 }

297 /// Helper method to change the active setpoint
298 void changeSetpoint( float delta )
299 {
300 // Update Cooling Setpoint
301 if ( m_coolingSetpoingActive )
302 {
303 g_uiCoolingSetpoint += delta;
304
305 // Enforce cooling range
306 if ( g_uiCoolingSetpoint > OPTION_STORM_DM_MP_SETPOINTS_MAX_COOLING )
307 {
308 g_uiCoolingSetpoint = OPTION_STORM_DM_MP_SETPOINTS_MAX_COOLING;
309 }
310 else if ( g_uiCoolingSetpoint < OPTION_STORM_DM_MP_SETPOINTS_MIN_COOLING )
311 {
312 g_uiCoolingSetpoint = OPTION_STORM_DM_MP_SETPOINTS_MIN_COOLING;
313 }
314
315 // Enforce Cooling is ALWAYS higher than the heating setpoint
316 if ( g_uiCoolingSetpoint <= g_uiHeatingSetpoint )
317 {
318 // Note: The setpoint ranges are such that the +1 operation does not violate the other setpoint's range
319 g_uiHeatingSetpoint = g_uiCoolingSetpoint - 1.0F;
320 }
321 }
322
323 // Update Heating Setpoint
324 else
325 {
326 g_uiHeatingSetpoint += delta;
327
328 // Enforce heating range
329 if ( g_uiHeatingSetpoint > OPTION_STORM_DM_MP_SETPOINTS_MAX_HEATING )
330 {
331 g_uiHeatingSetpoint = OPTION_STORM_DM_MP_SETPOINTS_MAX_HEATING;
332 }
333 else if ( g_uiHeatingSetpoint < OPTION_STORM_DM_MP_SETPOINTS_MIN_HEATING )
334 {
335 g_uiHeatingSetpoint = OPTION_STORM_DM_MP_SETPOINTS_MIN_HEATING;
336 }
337
338 // Enforce Cooling is ALWAYS higher than the heating setpoint
339 if ( g_uiCoolingSetpoint <= g_uiHeatingSetpoint )
340 {
341 // Note: The setpoint ranges are such that the +1 operation does not violate the other setpoint's range
342 g_uiCoolingSetpoint = g_uiHeatingSetpoint + 1.0F;
343 }
344 }
