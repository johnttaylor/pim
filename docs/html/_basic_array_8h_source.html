308  {
309  // Lock the Model Point
310  Cpl::Dm::ModelPointCommon_::m_modelDatabase.lock_();
311  uint16_t seqnum = BasicArray<ELEMTYPE>::m_seqNum;
312  int8_t valid = BasicArray<ELEMTYPE>::m_validState;
313  bool locked = BasicArray<ELEMTYPE>::m_locked;
314 
315  // Start the conversion
316  JsonDocument& doc = Cpl::Dm::ModelPointCommon_::beginJSON( valid, locked, seqnum, verbose );
317 
318  // Construct the 'val' array object
319  if ( Cpl::Dm::ModelPointCommon_::IS_VALID( valid ) )
320  {
321  JsonObject obj = doc.createNestedObject( "val" );
322  obj["start"] = 0;
323  JsonArray arr = obj.createNestedArray( "elems" );
324  for ( size_t i = 0; i < BasicArray<ELEMTYPE>::m_data.numElements; i++ )
325  {
326  if ( m_decimal )
327  {
328  arr.add( BasicArray<ELEMTYPE>::m_data.elemPtr[i] );
329  }

333  s.format( "0x%llX", ( unsigned long long ) BasicArray<ELEMTYPE>::m_data.elemPtr[i] );
334  arr.add( (char*)s.getString() );
335  }
336  }
337  }
338 
339  // End the conversion
340  Cpl::Dm::ModelPointCommon_::endJSON( dst, dstSize, truncated, verbose );
341

349  {

352  {
353  if ( errorMsg )
354  {
355  *errorMsg = "'val' key/value pair is NOT an JSON object";
356  }
357  return false;
358  }
359 
360  // Check for embedded array
361  JsonArray elems = src["elems"];
362  if ( elems.isNull() )
363  {
364  if ( errorMsg )
365  {
366  *errorMsg = "'val' key/value pair is missing the embedded 'elems' array";
367  }
368  return false;
369  }
370

372  size_t startIdx = src["start"];
373 
374  // Check for exceeding array limits
375  size_t numElements = elems.size();
376  if ( numElements > BasicArray<ELEMTYPE>::m_data.numElements )
377  {
378  if ( errorMsg )
379  {
380  errorMsg->format( "Number of array elements (%lu) exceeds the MP's element count (%lu)", src.size(), BasicArray<ELEMTYPE>::m_data.numElements );
381  }
382  return false;
383  }
384 
385  // Attempt to parse the value key/value pair (as a simple numeric)
386  ELEMTYPE* tempArray = (ELEMTYPE*)ModelDatabase::g_tempBuffer_;
387  for ( size_t i = 0; i < numElements; i++ )
388  {
389  // Attempt to parse the value as simple numeric
390  if ( m_decimal )
391  {
392  tempArray[i] = elems[i];
393  }
394 
395  // Attempt to parse the value as HEX string
396  else
397  {

400  if ( Cpl::Text::a2ull( value, val, 16 ) == false )
401  {

409  }
410  }

413  retSequenceNumber = BasicArray<ELEMTYPE>::write( tempArray, numElements, lockRequest, startIdx );
414  return true;
