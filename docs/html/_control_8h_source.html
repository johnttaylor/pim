31/** This concrete class provides the Component wrapper that is responsible for

41 /** This abstract class defines the interface for the Control Component to
42 invoke the control logic/processing on HVAC system with a specific equipment
43 configuration and/or operating mode (i.e. heating or cooling)
44 */
45 class Equipment
46 {
47 public:
48 /** This structure holds all of the data/info/configuration/etc. that
49 is needed by Equipment instances to executing their business logic.
50 A struct is used to reduce the number of arguments passed to the
51 various methods.
52 */
53 typedef struct
54 {
55 float pvOut; //!< IN: The process variable that represents the 'load' on the entire system
56 Storm::Type::SystemConfig_T systemConfig; //!< IN: Current System configuration (based on current operating mode)
57 Cpl::System::ElapsedTime::Precision_T currentTick; //!< IN: Value represents the current time of the main() loop iteration, i.e.the time of the start of the loop iteration.
58 Cpl::System::ElapsedTime::Precision_T currentInterval; //!< IN: Value is the deterministic interval boundary / time of when the component is logically executing.
59 Storm::Type::EquipmentTimes_T equipmentBeginTimes; //!< IN: The starting time for equipment's on/off cycles
60 Storm::Type::WhiteBox_T whiteBox; //!< IN: Collection of WhiteBox testing flags/settings/etc.
61 Storm::Type::VirtualOutputs_T vOutputs; //!< IO: The System's HVAC output values
62 Storm::Type::CycleInfo_T cycleInfo; //!< IO: Information about the current off/on cycle
63 bool systemOn; //!< IO: Is set to true if the system is actively conditioning the space (is NOT the same as is-the-equipment-on)
64 } Args_T;
65
66 public:
67 /** This method will be called on a periodic basis (as determined by the
68 calling Control Component instance) to perform active conditioning.
69 This method will only be called when the current operating mode 'matches'
70 the operating configuration of the Equipment instance.
71
72 The method returns true if no error(s) occurred; else false is returned.
73 Note: If false is return, then the Component's MP HVAC outputs will be
74 updated to an "all off state" (the Outdoor SOV output will
75 not be changed if/when this happens).
76 */
77 virtual bool executeActive( Args_T& args ) noexcept = 0;
78
79 /** This method will be called on a periodic basis (as determined by the
80 calling Control Component instance) to perform active conditioning.
81 This method will only be called when the current operating mode DOES
82 NOT MATCH the operating configuration of the Equipment instance.
83
84 The method returns true if no error(s) occurred; else false is returned.
85
86 Note: This method NEVER updates the Component's MP HVAC outputs.
87 */
88 virtual bool executeOff( Args_T& args ) noexcept = 0;
89
90 /** This method completes any/all initialization for the Equipment
91 instance. The method is called when the containing Component instance's
92 start() method is called. This method return true when successfully;
93 else false is returned.
94 */
95 virtual bool start( Cpl::System::ElapsedTime::Precision_T& intervalTime ) noexcept = 0;
96
97 /** This method is used to 'reset' the equipment. This method is intended
98 to be called whenever the System Type changes
99 */
100 virtual void reset() noexcept = 0;
101
102 /** Returns the Equipment instance's operating mode
103 */
104 virtual Storm::Type::OperatingMode getOperatingMode() const noexcept = 0;
105
106 public:
107 /// Virtual destructor
108 virtual ~Equipment() {}
109 };
