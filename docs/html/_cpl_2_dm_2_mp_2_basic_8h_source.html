164 public:
165  /// See Cpl::Dm::Point.
166  bool toJSON( char* dst, size_t dstSize, bool& truncated, bool verbose=true ) noexcept
167  {
168  // Get a snapshot of the my data and state
169  Cpl::Dm::ModelPointCommon_::m_modelDatabase.lock_();
170  ELEMTYPE value = Basic<ELEMTYPE>::m_data;
171  uint16_t seqnum = Basic<ELEMTYPE>::m_seqNum;
172  int8_t valid = Basic<ELEMTYPE>::m_validState;
173  bool locked = Basic<ELEMTYPE>::m_locked;
174  Cpl::Dm::ModelPointCommon_::m_modelDatabase.unlock_();
175 
176  // Start the conversion
177  JsonDocument& doc = Cpl::Dm::ModelPointCommon_::beginJSON( valid, locked, seqnum, verbose );
178 
179  // Construct the 'val' key/value pair (as a simple numeric)
180  if( Cpl::Dm::ModelPointCommon_::IS_VALID( valid ) )
181  {
182  if( m_decimal )
183  {
184  doc["val"] = value;
185  }
186 
187  // Construct the 'val' key/value pair (as a HEX string)
188  else
189  {
190  Cpl::Text::FString<20> tmp;
191  tmp.format( "0x%llX", ( unsigned long long ) value );
192  doc["val"] = (char*) tmp.getString();
193  }
194  }
195 
196  // End the conversion
197  Cpl::Dm::ModelPointCommon_::endJSON( dst, dstSize, truncated, verbose );
198  return true;
199  }

201  /// See Cpl::Dm::Point.
202  bool fromJSON_( JsonVariant& src, Cpl::Dm::ModelPoint::LockRequest_T lockRequest, uint16_t& retSequenceNumber, Cpl::Text::String* errorMsg ) noexcept
203  {
204  ELEMTYPE newValue = 0;
205 
206  // Attempt to parse the value key/value pair (as a simple numeric)
207  if( m_decimal )
208  {
209  ELEMTYPE checkForError = src | (ELEMTYPE) 2;
210  newValue = src | (ELEMTYPE) 1;
211  if( newValue == (ELEMTYPE) 1 && checkForError == (ELEMTYPE) 2 )
212  {
213  if( errorMsg )
214  {
215  *errorMsg = "Invalid syntax for the 'val' key/value pair";
216  }
217  return false;
218  }
219  }
220 
221  // Attempt to parse the value as HEX string
222  else
223  {
224  const char* val = src;
225  unsigned long long value;
226  if( Cpl::Text::a2ull( value, val, 16 ) == false )
227  {
228  if( errorMsg )
229  {
230  *errorMsg = "Invalid syntax for the 'val' key/value pair";
231  }
232  return false;
233  }
234 
235  newValue = (ELEMTYPE) value;
236  }
237 
238  retSequenceNumber = Basic<ELEMTYPE>::write( newValue, lockRequest );
239  return true;
240  }
